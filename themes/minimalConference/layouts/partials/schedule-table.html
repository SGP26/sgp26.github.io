{{- /* Renders a CSS Grid schedule from an array of course/session maps. Each item may have: title, category (course|coffee|lunch|welcome), timestamp (ISO 8601), duration (Go duration string), speakers ([]{name,url,affiliation}), abstract Parameters: courses – required slice of course/session maps */ -}} {{- $courses := .courses | default slice -}} {{- /* Collect unique dates, sort them, then assign column numbers */ -}} {{- $uniqueDates := slice -}} {{- $dateCols := dict -}} {{- range $c := $courses -}} {{- if and (.timestamp) (ne .timestamp "") -}} {{- $dKey := (time .timestamp).Format "2006-01-02" -}} {{- if not (in $uniqueDates $dKey) -}} {{- $uniqueDates = $uniqueDates | append $dKey -}} {{- end -}} {{- end -}} {{- end -}} {{- $sortedDates := sort $uniqueDates -}} {{- $colIdx := 2 -}} {{- range $sortedDates -}} {{- $dateCols = merge $dateCols (dict . $colIdx) -}} {{- $colIdx = add $colIdx 1 -}} {{- end -}} {{- /* Build entries and sort by time */ -}} {{- $entries := slice -}} {{- range $c := $courses -}} {{- if and (.timestamp) (ne .timestamp "") -}} {{- if not (.duration) -}}{{- errorf "%s: missing required 'duration' field" .title -}}{{- end -}} {{- $ts := time .timestamp -}} {{- $entries = $entries | append (dict "item" $c "time" ($ts.Format "15:04") "end" (($ts.Add (time.ParseDuration .duration)).Format "15:04") "dayCol" (index $dateCols ($ts.Format "2006-01-02")) "hasAbstract" (isset $c "abstract") ) -}} {{- end -}} {{- end -}} {{- $entries = sort $entries "time" -}}
